```go
type Watcher struct {
    source    EventSource
    logger    *slog.Logger
    eventCh   chan SandboxEvent  // output channel
}

type SandboxEvent struct {
    AppName     string
    ContainerID string
    EventType   string  // "container_started", "container_exited", "container_oom"
    ExitCode    string
    Time        time.Time
    OOMKilled   bool
}

func NewWatcher(source EventSource, logger *slog.Logger) *Watcher

func (w *Watcher) Watch(ctx context.Context) <-chan SandboxEvent
```

Watch() implementation:
- Starts a goroutine that calls source.EventsStream(ctx, since)
- On event: extract container name, skip if no "forge-" prefix, strip prefix for appName
  - Map "die" -> "container_exited", "oom" -> "container_oom", "start" -> "container_started"
  - Update `since` to event.Time
  - Send SandboxEvent on eventCh
- On error: log warning, apply exponential backoff (1s base, 2x multiplier, 30s max), reconnect with Since=last event timestamp
- On ctx.Done(): close eventCh, return

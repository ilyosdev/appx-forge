```go
type RegisterRequest struct {
    Hostname       string  `json:"hostname"`
    TailscaleIP    string  `json:"tailscale_ip"`
    AgentListenPort int    `json:"agent_listen_port"`
    CapacityMB     int     `json:"capacity_mb"`
    CapacityCPU    float64 `json:"capacity_cpu"`
    AgentVersion   string  `json:"agent_version"`
}

type RegisterResponse struct {
    NodeID                   string `json:"node_id"`
    AgentToken               string `json:"agent_token"`
    HeartbeatIntervalSeconds int    `json:"heartbeat_interval_seconds"`
}

type HeartbeatRequest struct {
    UsedMB            int `json:"used_mb"`
    RunningContainers int `json:"running_containers"`
}

type Command struct {
    ID             string          `json:"id"`
    Type           string          `json:"type"`
    SandboxID      string          `json:"sandbox_id"`
    Payload        json.RawMessage `json:"payload"`
    IssuedAt       time.Time       `json:"issued_at"`
    TimeoutSeconds int             `json:"timeout_seconds"`
}

type CommandsResponse struct {
    Commands []Command `json:"commands"`
}

type AckRequest struct {
    Status string      `json:"status"`
    Error  string      `json:"error,omitempty"`
    Result interface{} `json:"result,omitempty"`
}

type EventReport struct {
    SandboxID   string      `json:"sandbox_id"`
    EventType   string      `json:"event_type"`
    ContainerID string      `json:"container_id,omitempty"`
    ExitCode    int         `json:"exit_code,omitempty"`
    Payload     interface{} `json:"payload,omitempty"`
}
```

```go
type Client struct {
    baseURL    string
    httpClient *http.Client
    nodeID     string
    token      string
    regReq     RegisterRequest  // saved for re-registration
    logger     *slog.Logger
    mu         sync.RWMutex     // protects nodeID and token
}

func NewClient(baseURL string, regReq RegisterRequest, logger *slog.Logger) *Client

func (c *Client) Register(ctx context.Context) (*RegisterResponse, error)
func (c *Client) Heartbeat(ctx context.Context, req HeartbeatRequest) error
func (c *Client) PollCommands(ctx context.Context, waitSeconds int) ([]Command, error)
func (c *Client) AckCommand(ctx context.Context, cmdID string, ack AckRequest) error
func (c *Client) ReportEvent(ctx context.Context, event EventReport) error

// Internal helpers:
func (c *Client) doRequest(ctx, method, path, body) (*http.Response, error)
// - Adds Authorization: Bearer header
// - On 401: call Register(), retry once
// - On 404 (heartbeat only): call Register()
```

Key implementation details:
- Register: POST to /v1/nodes/register, store nodeID + token on success
- Heartbeat: POST to /v1/nodes/{nodeID}/heartbeat, on 404 re-register
- PollCommands: GET to /v1/agents/{nodeID}/commands?wait={waitSeconds}, HTTP timeout = (waitSeconds+5)*time.Second
- AckCommand: POST to /v1/agents/{nodeID}/commands/{cmdID}/ack
- ReportEvent: POST to /v1/agents/{nodeID}/events (fire-and-forget, retry once on error)
- All methods except Register include Authorization: Bearer {token}

| Threat ID | Category | Component | Disposition | Mitigation Plan |

| T-02-10 | Spoofing | controlclient | mitigate | Bearer token auth on all requests. Token scoped per node. Re-register on 401 to refresh token. |
